```
int red = 13;
int blue = 5;
int buzzer = 2;

void setup() {
  pinMode(red, OUTPUT);
  pinMode(blue, OUTPUT);
  pinMode(buzzer, OUTPUT);
}

void loop() {
  digitalWrite(red, HIGH);
  digitalWrite(blue, LOW);
  tone(buzzer, 740);   
  delay(1000);


  digitalWrite(red, LOW);
  digitalWrite(blue, HIGH);
  tone(buzzer, 523);   
  delay(1000);

}
```
