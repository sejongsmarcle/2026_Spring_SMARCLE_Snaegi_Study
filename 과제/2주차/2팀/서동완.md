int buzzer = 8;
int r = 9;
int b = 10;

void setup() {
  pinMode(buzzer, OUTPUT);
  pinMode(r, OUTPUT);
  pinMode(b, OUTPUT);
}

void loop() {
  tone(buzzer, 740);
  digitalWrite(r, HIGH);
  digitalWrite(b, LOW);
  delay(2000);


  noTone(buzzer);
  digitalWrite(r, LOW);
  delay(500);

  tone(buzzer, 523);
  digitalWrite(b, HIGH);
  digitalWrite(r, LOW);
  delay(2000);

  noTone(buzzer);
  digitalWrite(b, LOW);
  delay(500);
}
